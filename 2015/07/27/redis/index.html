redis

Redis是一个key-value存储系统。和Memcached类似，它支持存储的value类型相对更多，包括string(字符串)、list(链表)、set(集合)、zset(sorted set —有序集合)和hash（哈希类型）。这些数据类型都支持push/pop、add/remove及取交集并集和差集及更丰富的操作，而且这些操作都是原子性的。在此基础上，redis支持各种不同方式的排序。与memcached一样，为了保证效率，数据都是缓存在内存中。区别的是redis会周期性的把更新的数据写入磁盘或者把修改操作写入追加的记录文件，并且在此基础上实现了master-slave

一、 安装
先下载需要的安装
官方下载：http://redis.io/download 可以根据需要下载不同版本
windows版：https://github.com/mythz/redis-windows
附件是我下载的windows版本，可直接使用，将redis包解压到磁盘中，然后打开DOS窗口切换到解压的路径，进入redis目录后开启服务（注意加上redis.conf）
输入：redis-server.exe redis.conf
启动后可通过关闭该窗口来关闭redis服务

二、 使用

  1. 命令行使用
    在开始redis服务的情况下，在打开一个DOS窗口，并切换到redis目录，执行redis-cli.exe，如果需要修改ip和端口，在redis-cli.exe后增加-h 127.0.0.1 –p 6379
    完整的是：redis-cli.exe -h 127.0.0.1 –p 6379
    本地默认的ip就是127.0.0.1,端口默认的也是6379，所以可以直接执行redis-cli.exe。
    执行redis-cli.exe后可以直接输入set key value命令来设置值，通过get key来获取值，当然还有其它的命令来设置不同结构数据，如mset key1 value1 key2 value2 连续设置两个key-value对值
    获取是通过 mset key1 key2得到的value是一个List的集合 。其它的命令不一一列举，后面在Java中会有讲到。
    有意思的是如果一个事务中的某个命令执行出错了，事务中其它的命令照样执行，不会有影响。Redis的事务没有关系数据库事务提供的回滚（rollback） 功能，我们只能自己手动执行命令将数据库复原回事务执行前的状态等。另外咱们常用的都是在程序中执行，Java是会有语法检错的。

  2. Java中使用
    Redis的Java首选开发包是jedis
    Jedis的jar下载地址：http://maven.outofmemory.cn/redis.clients/jedis/2.5.2/
    将下载好的jedis jar包放到自己建好的工程包目录下，引入一些基础的Apache和jdk包即可。
    然后写一个单元测试类JedisUtilTest.java，如下：

    import org.junit.Before;
    import org.junit.Test;
    import redis.clients.jedis.Jedis;
    import redis.clients.jedis.JedisPool;
    import redis.clients.jedis.JedisPoolConfig;
    import java.util.HashMap;
    import java.util.Iterator;
    import java.util.List;
    import java.util.Map;

public class JedisUtilTest { 

JedisPool pool; 
Jedis jedis; 
@Before 
public void setUp() { 
    pool = new JedisPool(new JedisPoolConfig(), "127.0.0.1");  
    jedis = pool.getResource();  
} 
@Test 
public void testGet(){ 
    System.out.println(jedis.get("name")); 
} 

/** 
 * Redis操作字符串
 */ 
@Test 
public void testBasicString(){
        //-----添加数据---------- 
        jedis.set("name","lily");//向key-->name中放入了value-->lily 
        System.out.println("set----"+jedis.get("name"));//执行结果：lily 
        //-----修改数据----------- 
        //1、在原来基础上修改 
        jedis.append("name","莉莉");   //很直观，类似map 将莉莉 append到已经有的value之后 
        System.out.println("append----"+jedis.get("name"));//执行结果:lily莉莉  
        //2、直接覆盖原来的数据 
        jedis.set("name","莉莉"); 
        System.out.println(jedis.get("name"));//执行结果：莉莉 
        //删除key对应的记录
        jedis.del("name"); 
        System.out.println("delete----"+jedis.get("name"));//执行结果：null     
        //批量操作 
        jedis.mset("name","lily","jarorwar","莉莉");            
        System.out.println("mset----"+jedis.mget("name","jarorwar")); 
} 

/** 
 * jedis操作Map 
 */ 
@Test 
public void testMap(){ 
    Map<String,String> user=new HashMap<String,String>(); 
    user.put("name","莉莉"); 
    user.put("pwd","password"); 
    jedis.hmset("user",user);
    //取出user中的name，执行结果:[莉莉]-->注意结果是一个泛型的List 
    //第一个参数是存入redis中map对象的key，后面跟的是放入map中的对象的key，后面的key可以跟多个，是可变参数 
    List<String> rsmap = jedis.hmget("user", "name"); 
    System.out.println("hmset----"+rsmap);  
    //删除map中的某个键值 
    jedis.hdel("user","pwd"); 
    System.out.println(jedis.hmget("user", "pwd")); //因为删除了，所以返回的是null 
    System.out.println(jedis.hlen("user")); //返回key为user的键中存放的值的个数1 
    System.out.println(jedis.exists("user"));//是否存在key为user的记录 返回true
    System.out.println(jedis.hkeys("user"));//返回map对象中的所有key  [pwd, name] 
    System.out.println(jedis.hvals("user"));//返回map对象中的所有value  [minxr, password] 
    Iterator<String> iter=jedis.hkeys("user").iterator(); 
    while (iter.hasNext()){ 
        String key = iter.next();            
        System.out.println(key+":"+jedis.hmget("user",key));
    }     
} 

/** 
 * jedis操作List 
 */ 
@Test 
public void testList(){ 
        //开始前，先移除所有的内容 
        jedis.del("list"); 
        System.out.println(jedis.lrange("list",0,-1)); 
        //先向key java framework中存放三条数据 
        jedis.lpush("list","lily"); 
        jedis.lpush("list","25"); 
        jedis.lpush("list","women"); 
        //再取出所有数据jedis.lrange是按范围取出，
        //第一个是key，第二个是起始位置，第三个是结束位置，jedis.llen获取长度 -1表示取得所有
        System.out.println(jedis.lrange("list",0,-1)); 
    }

/** 
 * jedis操作Set 
 */ 
@Test 
public void testSet(){ 
        //添加 
        jedis.sadd("sname","minxr"); 
        jedis.sadd("sname","jarorwar"); 
        jedis.sadd("sname","莉莉"); 
        jedis.sadd("sanme","noname"); 
        //移除noname 
        jedis.srem("sname","noname"); 
        System.out.println(jedis.smembers("sname"));
        //获取所有加入的value 
        System.out.println(jedis.sismember("sname", "minxr"));
        //判断 minxr 是否是sname集合的元素 
        System.out.println(jedis.srandmember("sname")); 
        System.out.println(jedis.scard("sname"));//返回集合的元素个数 
    }

@Test 
public void test() throws InterruptedException { 
        //keys中传入的可以用通配符
        System.out.println(jedis.keys("*")); //返回当前库中所有的key
        System.out.println(jedis.keys("*name"));//返回的sname   [sname, name] 
        System.out.println(jedis.del("sanmdde"));//删除key为sanmdde的对象  删除成功返回1 删除失败（或者不存在）返回 0 
        System.out.println(jedis.ttl("sname"));//返回给定key的有效时间，如果是-1则表示永远有效 
        jedis.setex("timekey", 10, "min");//通过此方法，可以指定key的存活（有效时间） 时间为秒 
        Thread.sleep(5000);//睡眠5秒后，剩余时间将为<=5  
        System.out.println(jedis.ttl("timekey"));   //输出结果为5
        jedis.setex("timekey", 1, "min");        //设为1后，下面再看剩余时间就是1了
        System.out.println(jedis.ttl("timekey"));  //输出结果为1
        System.out.println(jedis.exists("key"));//检查key是否存在      
        System.out.println(jedis.rename("timekey","time")); 
        System.out.println(jedis.get("timekey"));//因为移除，返回为null 
        System.out.println(jedis.get("time")); //因为将timekey 重命名为time 所以可以取得值 min 
        //jedis 排序 
        //注意，此处的rpush和lpush是List的操作。是一个双向链表（但从表现来看的）  
        jedis.del("a");//先清除数据，再加入数据进行测试
        jedis.rpush("a", "1"); 
        jedis.lpush("a","6"); 
        jedis.lpush("a","3"); 
        jedis.lpush("a","9"); 
        System.out.println(jedis.lrange("a",0,-1));// [9, 3, 6, 1] 
        System.out.println(jedis.sort("a")); //[1, 3, 6, 9]  //输入排序后结果
        System.out.println(jedis.lrange("a",0,-1)); 
    }

}

注意：我在测试的过程中遇到了一个问题，弄了好长时间没有解决，就是写测试类初始化时JedisPool pool=new JedisPool(new JedisPoolConfig(), “127.0.0.1”);
总是报初始化错误，原因是找不到GenericObjectPoolConfig类，上网上搜了一下，说是没有引入commons-pool包，但是我发现工程里已经有commons-pool-1.6.jar，然后我就反编译了commons-pool-1.6.jar，发现里面根本就没有GenericObjectPoolConfig类。后来又查找，大家都说是jedis版本的问题，我就一顿的降版本，将了好几个都没使，网上也有和我遇到一样问题的，最后没有给出解决方案。最后我换了个思路，查查commons-pool有没有更高版本的，找到commons-pool2-2.2.jar更高版本后居然好使了。
